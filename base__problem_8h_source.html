472 run(int argc, char **argv)
473 {
474  try
475  {
476  using namespace dealii;
477 
478  Utilities::MPI::MPI_InitFinalize mpi_initialization(argc, argv);
479  std::string par_name = "";
480  if (argc > 1)
481  par_name = argv[1];
482 
483  if (Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
484  deallog.depth_console(2);
485  else
486  deallog.depth_console(0);
487 
488  ProblemType base_problem;
489  base_problem.initialize(par_name);
490  base_problem.run();
491  }
492  catch (std::exception &exc)
493  {
494  std::cerr << std::endl
495  << std::endl
496  << "----------------------------------------------------"
497  << std::endl;
498  std::cerr << "Exception on processing: " << std::endl
499  << exc.what() << std::endl
500  << "Aborting!" << std::endl
501  << "----------------------------------------------------"
502  << std::endl;
503 
504  return 1;
505  }
506  catch (...)
507  {
508  std::cerr << std::endl
509  << std::endl
510  << "----------------------------------------------------"
511  << std::endl;
512  std::cerr << "Unknown exception!" << std::endl
513  << "Aborting!" << std::endl
514  << "----------------------------------------------------"
515  << std::endl;
516  return 1;
